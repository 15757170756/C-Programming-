8.C
#include <stdlib.h>

int main() 
{
	int x, y;
	for (y = 1, x = 1; y <= 50; y++) {
		if (x >= 10) break;
		if (x % 2 == 1) {
			x += 5;
			continue;
		}
		x -= 3;
	}
	printf("%d\n", y);
	return 0;
}
顺序执行下来，要注意，for语句中的continue语句，表达式语句的执行不会跳过，
而while语句则会跳过，和上面一模一样的while语句程序：
#include <stdlib.h>

int main() 
{
	int x, y;
	x = 1;
	y = 1;
	while (y < 50) {
		if (x >= 10) break;
		if (x % 2 == 1) {
			x += 5;
			continue;
		}
		x -= 3;
		y++;
	}
	printf("%d\n", y);
	return 0;
}
执行结果是y=3

9.C
#include <stdlib.h>

int main() 
{
	int jc;
	for (int i = 2; i <= 10; i++)
		jc *= i;
	printf("%d\n", jc);

	return 0;
}
因为规定sizeof(int)为2 即int类型的变量最大为65536，而10！= 3628800，所以不行。

for (float jc = 1, int i = 2; i <= 10; i++, jc *= i);
是有语法问题，不同类型的变量定义时不能用逗号。

#include <stdlib.h>
int main() 
{
	float jc;
	for (int i = 2; i <= 10; jc *= i, i = i + 1);
		//jc *= i;
	printf("%d\n", jc);

	return 0;
}可以只用float类型

	int a = 1, b = 0;
	do {
		switch (a) {
		case 1: b = 1; break;
		case 2: b = 2; break;

		}
		b = a + b;
